#1. Напишите программу, которая принимает
# на вход цифру, обозначающую день недели, и
# проверяет, является ли этот день выходным.
#*Пример:*

#- 6 -> да
#- 7 -> да
#- 1 -> нет

print("Является день выходным?")
n = int(input())
if (n >= 1 and n <= 5):
    print("Рабочий день")
elif (n == 6):
    print("Выходной день")
elif (n == 7):
    print("Выходной день")
else:
    print("Ошибка, не существует данного дня")
